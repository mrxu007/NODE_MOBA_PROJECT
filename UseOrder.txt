// vue-cli常用指令
npm install -g @vue/cli          vue-cli脚手架全局安装
vue create [项目名称]             创建项目

//Git常用指令
git remote -v                   查看仓库地址
git remote add origin [地址]    添加仓库地址
git status                      查看当前版本库的状态
git add .                       添加工作区的文件
git commit -am “[备注]”         代码添加当本地版本库
git branch                      查看本地分支
git branch                      查看本地分支和远程分支
git branch [分支名]             创建分支
git checkout [分支名]           切换分支
git merge [分支名]              合并分支
git pull                        从远程端拉取代码
git push                        推送代码到远程端
